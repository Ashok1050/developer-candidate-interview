# Ruby program that takes a string as input and determines whether or # not the string is a palindrome. The string could be a word or a # phrase. The kicker... 
# Input Type: string
# Output Type: boolean
# scrub_string method ---- palindrome method 
# Input number is convert to string using scrub string

def scrub_string str
  str = str.to_s if !str.is_a?String
  str = str.gsub(/\W|\_/, "").downcase
end

def is_palindrome? str
  str = scrub_string(str)
  (str.length/2).times do |index_counter|
    if str[index_counter] != str[-(index_counter + 1)]
      return false
    end
  end
  true
end

def is_palindrome? str
  str = scrub_string(str)
  str == str.reverse ? true : false
end
